    Using Matchers

    Overview

    Matchers let you test values in expressive ways. All matchers support .not for negation, and async matchers are supported via resolves and rejects.

    Common Matchers

    MatcherDescriptionExample
    toBeStrict equality (===)expect(2 + 2).toBe(4);
    toEqualDeep equality (objects/arrays)expect({a:1}).toEqual({a:1});
    toBeNullChecks for nullexpect(null).toBeNull();
    toBeUndefinedChecks for undefinedexpect(undefined).toBeUndefined();
    toBeCloseToFloating point comparisonexpect(0.1 + 0.2).toBeCloseTo(0.3);
    toBeTruthyChecks for truthy valueexpect(1).toBeTruthy();
    toBeFalsyChecks for falsy valueexpect(0).toBeFalsy();
    toBeGreaterThanGreater thanexpect(10).toBeGreaterThan(5);
    toBeGreaterThanOrEqualGreater than or equalexpect(5).toBeGreaterThanOrEqual(5);
    toBeLessThanLess thanexpect(3).toBeLessThan(5);
    toBeLessThanOrEqualLess than or equalexpect(3).toBeLessThanOrEqual(3);
    toBeNaNChecks for NaNexpect(NaN).toBeNaN();
    toMatchRegex/string matchexpect('hello').toMatch(/ell/);
    toMatchObjectPartial object matchexpect({a:1,b:2}).toMatchObject({a:1});
    toContainArray/string containsexpect([1,2,3]).toContain(2);
    toHaveLengthArray/string lengthexpect([1,2]).toHaveLength(2);
    toHavePropertyObject property (optionally with value)expect({a:{b:2}}).toHaveProperty('a.b',2);
    toThrowFunction throws (optionally with error)expect(() => { throw new Error('fail'); }).toThrow('fail');
    toBeInstanceOfInstance checkexpect([]).toBeInstanceOf(Array);
    toBeArrayChecks for arrayexpect([]).toBeArray();
    toBeObjectChecks for objectexpect({}).toBeObject();
    toBeStringChecks for stringexpect('hi').toBeString();
    toBeNumberChecks for numberexpect(5).toBeNumber();
    toBeBooleanChecks for booleanexpect(true).toBeBoolean();
    toBeFunctionChecks for functionexpect(() => {}).toBeFunction();
    toBeDateChecks for dateexpect(new Date()).toBeDate();

    Negating Matchers

    All matchers support .not for negation:

    expect(2 + 2).not.toBe(5);
  expect('hello').not.toMatch('world');

    Custom Matchers

    Add your own matchers for custom assertions:

    const { addMatchers } = require('fauji/matchers');
  addMatchers({
    toBeDivisibleBy: (received, expected) => received % expected === 0
  });
  expect(10).toBeDivisibleBy(2);
    Custom matchers extend expect globally for all your tests.
